LITERALS 

	STRING 
		Cannot span multiple lines. 
		'\' is no longer an escape char.
	
	CHAR 
		Typical:
		'LETTER' (i.e. 'a', 'e', 'i', 'o', 'u')
	
	NUMBERS 
	
		INT 
			Three kinds:
			dec: 8, 	10, 	11	([0-9]*)
			oct: 010,	013, 	032	(0[0-7]*) 
			hex: 0x8,	0xa,	0xb	(0x[0-9A-Fa-f]*)
		
		REAL 
			Typical:
			{WHOLE_PART}.{FRAC_PART}{e || E}{maybe + || -}{EXP}
			(i.e  5.9834e-67, 100.43E+12)
