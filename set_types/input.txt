CONST
    high = 50;

TYPE
    LowNums  = 0..high;
    Giant    = (fee, fie, foe, fum);
    GSet     = SET OF Giant;
    NumSet   = SET OF LowNums;
    SetArray = ARRAY [Giant] OF NumSet;
    Array2d  = ARRAY [LowNums, Giant] OF fie..fum;

    RecordA = RECORD
                  fld1 : NumSet;
                  fld2 : SET OF Giant;
                  fld3 : SET OF 5..12;
                  fld4 : SetArray;
                  fld5 : Array2d;
                  fld6 : Giant;
              END;

    ArrayOfSet = ARRAY [Giant, -5..-1] OF SET OF Giant;

VAR
    gs   : GSet;
    s    : NumSet;
    t    : SET OF (alpha, beta, gamma);

BEGIN
END.
